324        self,

326        records: SQLRecord | Sequence[SQLRecord],
327    ) -> list[int] | None:
328        """
329        Insert records into a table.
330
331        Args:
332            table (SQLTable): The table to insert records into.
333            records (SQLRecord | Sequence[SQLRecord]): The records to insert.
334
335        Returns:
336            list[int] | None: The IDs of the inserted records, or None if no IDs are generated.
337        """
338        if isinstance(records, SQLRecord):
339            records = [records]
340
341        insert_statement = SQLInsertIntoStatement(self.dialect, table, records[0])
342        ids = []
343        for index, record in enumerate(records):
344            if index > 0 and insert_statement.template_parameters is not None:
345                for parameter, value in zip(
346                    insert_statement.template_parameters.keys(), record.values()
347                ):
348                    insert_statement.template_parameters[parameter] = value
349            cursor = self._execute_statement(insert_statement)
350            if cursor.description is not None:
351                row = cursor.fetchone()
352                if row is not None:
353                    ids.append(row[0])
354        return ids if len(ids) else None
